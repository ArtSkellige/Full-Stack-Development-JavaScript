let count = 0;

function cc(card) {
  // Update count based on card value
  if (card >= 2 && card <= 6) {
    count++;
  } else if (card >= 7 && card <= 9) {
    // same as above
  } else if (card === 10 || card === "J" || card === "Q" || card === "K" || card === "A") {
    count--;
  }
  
  // Return count + bet/hold
  if (count > 0) {
    return count + " Bet";
  } else {
    return count + " Hold";
  }
}
