function flattenPlaylists(input) {
  if (!Array.isArray(input)) return [];

  const result = [];

  for (let i = 0; i < input.length; i++) {
    for (let j = 0; j < input[i].length; j++) {
      const track = input[i][j];
      result.push({
        ...track,
        source: [i, j]
      });
    }
  }

  return result;
}

function scoreTracks(tracks) {
  return tracks.map(track => ({
    ...track,
    score: track.votes * 10 - Math.abs(track.bpm - 120)
  }));
}

function dedupeTracks(tracks) {
  const knownIds = new Set();

  return tracks.filter(track => {
    if (knownIds.has(track.trackId)) return false;
    knownIds.add(track.trackId);
    return true;
  });
}

function enforceArtistQuota(tracks, maxPerArtist) {
  const artistCounts = new Map();

  return tracks.filter(track => {
    const count = artistCounts.get(track.artist) ?? 0;
    if (count >= maxPerArtist) return false;
    artistCounts.set(track.artist, count + 1);
    return true;
  });
}

function buildSchedule(tracks) {
  return tracks.map((track, index) => ({
    slot: index + 1,
    trackId: track.trackId
  }))
}

function remixPlaylist(playlists, maxPerArtist) {
  const flat = flattenPlaylists(playlists);
  const scored = scoreTracks(flat);
  const deduped = dedupeTracks(scored);
  const quota = enforceArtistQuota(deduped, maxPerArtist);
  return buildSchedule(quota);
}
