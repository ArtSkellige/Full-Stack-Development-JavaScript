function validateForm() {
  return {
    "full-name": document.getElementById('full-name').value !== "",
    "email": /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(document.getElementById('email').value),
    "order-no": /^2024\d{6}$/.test(document.getElementById('order-no').value),
    "product-code": /^[a-zA-Z]{2}\d{2}-[a-zA-Z]\d{3}-[a-zA-Z]{2}\d$/.test(document.getElementById('product-code').value),
    "quantity": Number(document.getElementById('quantity').value) > 0,
    "complaints-group": Array.from(
      document.querySelectorAll('input[name="complaint"]')
    ).some(checkbox => checkbox.checked),
    "complaint-description": !document.getElementById('other-complaint').checked || document.getElementById('complaint-description').value.length >= 20,
    "solutions-group": Array.from(
      document.querySelectorAll('input[name="solutions"]')
    ).some(radio => radio.checked),
    "solution-description": !document.getElementById('other-solution').checked || document.getElementById('solution-description').value.length >= 20,
  };
}

function isValid(validationObject) {
  return Object.values(validationObject).every(value => value === true);
}

const form = document.getElementById('form');

form.addEventListener('submit', function(e) {
  e.preventDefault();
  const validation = validateForm();

  if (!isValid(validation)) {
    for (const key in validation) {
      if (validation[key] === false) {
        document.getElementById(key).style.borderColor = 'red';
      }
    }
  } else {
    document.getElementById('message-box').textContent = 'Form submitted successfully!';
    form.reset();
  }
});
