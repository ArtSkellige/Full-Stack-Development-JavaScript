** start of script.js **

const themes = [
  {
    name: "light",
    message: "Let there be light!☀️"
  },
  {
    name: "dark",
    message: "Lights out! Dark mode engaged🌙"
  },
  {
    name: "high-contrast",
    message: "Cranked up the contrast!👓"
  },
  {
    name: "greenwood",
    message: "Into the woods we go!🌲"
  },
  {
    name: "dark-fuchsia",
    message: "Pretty in pink... er, dark-fuchsia!🟪⬛️"
  }
]

const button = document.getElementById("theme-switcher-button");
const dropdown = document.getElementById("theme-dropdown");

button.addEventListener("click", () => {
  dropdown.hidden = !dropdown.hidden;

  button.setAttribute("aria-expanded", dropdown.hidden ? "false" : "true");
});

const themeItems = document.querySelectorAll('[role="menuitem"]');

themeItems.forEach(item => {
  item.addEventListener("click", () => {
    const themeName = item.id.replace("theme-", "");
    
    document.body.className = "";
    
    document.body.classList.add(`theme-${themeName}`);
    
    const selectedTheme = themes.find(theme => theme.name === themeName);
    
    const statusElement = document.querySelector(`[aria-live="polite"]`);
    statusElement.textContent = selectedTheme.message;

    dropdown.hidden = true;
    button.setAttribute("aria-expanded", "false");
  });
});

** end of script.js **
