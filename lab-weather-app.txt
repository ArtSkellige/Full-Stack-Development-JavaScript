** start of script.js **

async function getWeather(city) {
  try {
    const response = await fetch(
      `https://weather-proxy.freecodecamp.rocks/api/city/${city}`
    );
    const data = await response.json();
    return data;
  } catch (error) {
    console.log(error);
  }
}

async function showWeather(city) {
  const weatherData = await getWeather(city);

  if (!weatherData) {
    alert("Something went wrong, please try again later");
    return;
  }

  document.getElementById("weather-icon").src = 
    weatherData.weather[0].icon || "";
  document.getElementById("main-temperature").textContent = 
    weatherData.main.temp || "N/A";
  document.getElementById("feels-like").textContent = 
    weatherData.main.feels_like || "N/A";
  document.getElementById("humidity").textContent = 
    weatherData.main.humidity || "N/A";
  document.getElementById("wind").textContent = 
    weatherData.wind.speed || "N/A";
  document.getElementById("wind-gust").textContent = 
    weatherData.wind.gust || "N/A";
  document.getElementById("weather-main").textContent = 
    weatherData.weather[0].main || "N/A";
  document.getElementById("location").textContent = 
    weatherData.name || "N/A";
}

document.getElementById("get-weather-btn").addEventListener("click", () => {
  const citySelect = document.querySelector("select");
  const selectedCity = citySelect.value;

  if (selectedCity === "") {
    return;
  }

  showWeather(selectedCity);
});
