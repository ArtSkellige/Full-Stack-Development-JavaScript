function formatDateMMDDYYYY(FD) {
  const formatted = FD.toLocaleDateString('en-US', {
    year: 'numeric',
    month: 'numeric',
    day: '2-digit'
  });

  return `Formatted Date (MM/DD/YYYY): ${formatted}`;
}

function formatDateLong(FD) {
  const formatted = FD.toLocaleDateString('en-US', {
    year: 'numeric',
    month: 'long',
    day: 'numeric'
  });

  return `Formatted Date (Month Day, Year): ${formatted}`;
}
