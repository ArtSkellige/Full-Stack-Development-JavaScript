function dfs(graph, root) {
  const visited = [];
  const stack = [root];

  while (stack.length > 0) {
    const current = stack.pop();

    if (!visited.includes(current)) {
      visited.push(current);

      for (let i = 0; i < graph[current].length; i++) {
        if (graph[current][i] === 1 && !visited.includes(i)) {
          stack.push(i);
        }
      }
    }
  }
  return visited;
}
